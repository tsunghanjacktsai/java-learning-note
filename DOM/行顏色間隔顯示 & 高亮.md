# 行顏色間隔顯示 & 高亮

## 演示
```javascript
//table2.css
table {

	border: #FB0202 solid 2px;
}

table td {

	border: #2A09EF solid 1px;
	padding: 10px;
}
table th {

	border: #2A09EF solid 1px;
	padding: 10px;
	background-color: #F907D4;
}
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<link rel="stylesheet" type="text/css" href="CSS/table2.css">
	<title>Document</title>
	<style type="text/css">
		.one {

			background-color: #1CDFD1;
		}
		.two {

			background-color: #FCFA05;
		}
		.over {

			background-color: #35FF02;
		}
	</style>
	<script type="text/javascript">
		var name;
		function trColor() {

			/*
			 * 思路:
			 * 1. 因為要操作行的樣式，所以要先獲取行的樣式
			 * 2. 獲取所有被操作的行對象
			 * 3. 遍歷行並給每一行指定樣式
			 */
			 oTabNode = document.getElementById("info");

			 //表格中所有的行
			 var collTrNodes = oTabNode.rows;

			 //遍歷的時候從第二行開始遍歷
			 for(var x = 1; x < collTrNodes.length; x++) {

			 	if(x % 2 == 1) {

			 		collTrNodes[x].className = "one";
			 	} else {

			 		collTrNodes[x].className = "two";
			 	}

			 	//給每一行對象都添加兩個事件
			 	collTrNodes[x].onmouseover = function() {

			 		name = this.className;
			 		this.className = "over";
			 	}
			 	collTrNodes[x].onmouseout = function() {

			 		this.className = name;
			 	}
			 }

		}
		
		onload = function() {

		 	trColor();
		}

		/*
		function over(node) {

			name = node.className;
			node.className = "over";
		}

		function out(node) {

			node.className = name;
		}
		*/
	</script>
</head>
<body>
	<table id="info">
		<tr>
			<th>Name</th>
			<th><a href="javascript:void(0)">Age</a></th>
			<th>Address</th>
		</tr>
		<tr onmouseover="over(this)" onmouseout="out(this)">
			<td>Jack</td>
			<td>20</td>
			<td>Shanghai</td>
		</tr>
		<tr>
			<td>Jacky</td>
			<td>30</td>
			<td>Taiwan</td>
		</tr>
		<tr>
			<td>Vivian</td>
			<td>40</td>
			<td>Hong Kong</td>
		</tr>
		<tr>
			<td>Jervis</td>
			<td>50</td>
			<td>Beijing</td>
		</tr>
		<tr>
			<td>Mike</td>
			<td>60</td>
			<td>Suzhou</td>
		</tr>
	</table>
</body>
</html>
```
