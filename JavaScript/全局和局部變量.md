# 全局和局部變量

## 演示
```javascript
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Document</title>
</head>
<body>
	<script type="text/javascript">
		// for(var x = 0; x < 3; x++) { //在腳本片段中定義的變量，全局變量
		// 	document.write("x =" + x + "<br>");
		// }//全部變量

		// document.write("x===" + x);

		var x = 3;//全局變量 x

		function show(x) {//函數局部的變量 x
			var x = 6;
		}//局部變量

		show(x);

		document.write("x=======" + x + "<br>");//x = 3
	</script>
	
	<script type="text/javascript">
		// document.write("xx==" + x);
	</script>
</body>
</html>
```
